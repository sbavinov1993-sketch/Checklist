// ОНОВЛЕНА ФУНКЦІЯ ВІДКРИТТЯ МОДАЛЬНОГО ВІКНА function openModal(sectionId, taskId, taskData) { currentActiveTask = { sectionId, taskId, taskData }; document.getElementById('modal-task-title').innerText = taskData.text || taskData.title || "Деталі"; document.getElementById('modal-info').value = taskData.additionalInfo || ''; const imgPreview = document.getElementById('modal-photo-preview'); const img = document.getElementById('modal-img'); document.getElementById('modal-file').value = ''; // Динамічно створюємо або керуємо кнопкою видалення let btnDelete = document.getElementById('btn-delete-photo'); if (!btnDelete) { btnDelete = document.createElement('button'); btnDelete.id = 'btn-delete-photo'; btnDelete.className = "w-full mt-2 bg-red-100 hover:bg-red-200 text-red-600 font-medium py-1.5 rounded-lg transition text-sm hidden"; btnDelete.innerText = "❌ Видалити фотодоказ"; btnDelete.addEventListener('click', deletePhoto); imgPreview.parentNode.insertBefore(btnDelete, imgPreview.nextSibling); } if (taskData.photoUrl) { img.src = taskData.photoUrl; imgPreview.classList.remove('hidden'); btnDelete.classList.remove('hidden'); // Показуємо кнопку, якщо фото є } else { imgPreview.classList.add('hidden'); btnDelete.classList.add('hidden'); // Ховаємо, якщо фото немає } document.getElementById('modal').classList.remove('hidden'); } // НОВА ФУНКЦІЯ ДЛЯ МИТТЄВОГО ВИДАЛЕННЯ ФОТО async function deletePhoto() { if (!currentActiveTask) return; const { sectionId, taskId } = currentActiveTask; if (confirm("Ви впевнені, що хочете видалити цей фотодоказ?")) { const btnDelete = document.getElementById('btn-delete-photo'); btnDelete.innerText = "Видалення..."; btnDelete.disabled = true; try { // Просто затираємо поле фото в базі Firestore await updateDoc(doc(db, `sections/${sectionId}/tasks/${taskId}`), { photoUrl: "" }); // Оновлюємо інтерфейс у модалці document.getElementById('modal-photo-preview').classList.add('hidden'); btnDelete.classList.add('hidden'); // Оновлюємо дані в поточному об'єкті currentActiveTask.taskData.photoUrl = ""; } catch (err) { alert("Помилка видалення: " + err.message); } finally { btnDelete.innerText = "❌ Видалити фотодоказ"; btnDelete.disabled = false; } } } // ТРОХИ ОНОВЛЕНЕ ЗБЕРЕЖЕННЯ (додано очищення інпуту) async function saveModalData() { if (!currentActiveTask) return; const { sectionId, taskId } = currentActiveTask; const btnSave = document.getElementById('btn-save'); btnSave.innerText = "Зберігання..."; btnSave.disabled = true; const updatedData = { additionalInfo: document.getElementById('modal-info').value }; const fileInput = document.getElementById('modal-file'); if (fileInput.files.length > 0) { try { updatedData.photoUrl = await resizeAndConvertToBase64(fileInput.files[0]); } catch (error) { alert("Помилка фото: " + error.message); } } try { await updateDoc(doc(db, `sections/${sectionId}/tasks/${taskId}`), updatedData); closeModal(); } catch (err) { alert("Помилка Firestore: " + err.message); } finally { btnSave.innerText = "Зберегти зміни"; btnSave.disabled = false; } }
